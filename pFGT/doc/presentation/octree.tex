
\begin{frame}
\frametitle{Expansion or Truncation?}
\begin{itemize}
\item Many sources per FGT box $-$ Expansion
\newline
\item Few sources per FGT box $-$ Truncation
\newline
\item Non-uniform distributions $-$ Hybrid
\end{itemize}
\end{frame}
